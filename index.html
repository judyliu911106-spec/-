// 新增一個輕微震動的輔助函式 function triggerVibration(type = "light") { if (navigator.vibrate) { if (type === "light") { navigator.vibrate(15); // 短促輕微震動 (按鍵用) } else if (type === "success") { navigator.vibrate([40, 40, 40]); // 連續快震 (成功用) } } } // 修改按鍵點擊函式，加入震動 function pressKey(key) { triggerVibration("light"); // ✨ 觸發按鍵視覺/觸覺回饋 if (key === 'C') { currentAmount = currentAmount.length > 1 ? currentAmount.slice(0, -1) : "0"; } else { currentAmount = currentAmount === "0" ? key : currentAmount + key; } document.getElementById('tx-amount').innerText = currentAmount; } // 修改選擇分類函式，加入震動 function selectCat(element, cat) { triggerVibration("light"); // ✨ 觸發分類按鍵回饋 chosenCategory = cat; document.getElementById('selected-cat-text').innerText = cat; document.querySelectorAll('.cat-btn').forEach(btn => btn.classList.remove('selected')); element.classList.add('selected'); } // 修改送出成功的提示 function submitRecord() { if (!chosenCategory) { alert("請先選擇一個可愛的分類唷！🌸"); return; } if (currentAmount === "0") { alert("請輸入金額！"); return; } const payload = { date: document.getElementById('tx-date').value, category: chosenCategory, item: document.getElementById('tx-item').value, amount: currentAmount }; fetch(GAS_URL, { method: "POST", body: JSON.stringify(payload) }) .then(res => res.json()) .then(res => { triggerVibration("success"); // ✨ 成功時發出歡樂的連震回饋！ alert(res.message); location.reload(); }); }
